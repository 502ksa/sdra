<script>
const duas = ["اللهم ارزقني التوفيق","اللهم اغفر لي","اللهم اهدني"];
const azkar = ["سبحان الله","الحمد لله","الله أكبر"];

function changeDua(){
document.getElementById("dua").innerText = duas[Math.floor(Math.random()*duas.length)];
}
function changeZekr(){
document.getElementById("zekr").innerText = azkar[Math.floor(Math.random()*azkar.length)];
}

// تسبيح
let count = localStorage.getItem("tasbeeh") || 0;
document.getElementById("count").innerText = count;

function inc(){
count++;
localStorage.setItem("tasbeeh",count);
document.getElementById("count").innerText = count;
}
function reset(){
count=0;
localStorage.setItem("tasbeeh",count);
document.getElementById("count").innerText = count;
}

// صوت
function toggleAudio(){
let a = document.getElementById("audio");
a.paused ? a.play() : a.pause();
}

// مواقيت الصلاة
function getLocation(){
if(navigator.geolocation){
navigator.geolocation.getCurrentPosition(pos=>{
let lat = pos.coords.latitude;
let lon = pos.coords.longitude;

fetch(`https://api.aladhan.com/v1/timings?latitude=${lat}&longitude=${lon}&method=4`)
.then(res=>res.json())
.then(data=>{
let t = data.data.timings;

document.getElementById("city").innerText = "حسب موقعك";

let table = `
<tr><td>الفجر</td><td>${t.Fajr}</td></tr>
<tr><td>الظهر</td><td>${t.Dhuhr}</td></tr>
<tr><td>العصر</td><td>${t.Asr}</td></tr>
<tr><td>المغرب</td><td>${t.Maghrib}</td></tr>
<tr><td>العشاء</td><td>${t.Isha}</td></tr>
`;
document.getElementById("prayerTable").innerHTML = table;
});
});
}
}
getLocation();
</script>
